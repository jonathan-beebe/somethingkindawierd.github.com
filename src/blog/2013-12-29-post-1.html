$.fn.scrollLock = function () {
  return $(this).on("DOMMouseScroll mousewheel", function (h) {
    var g = $(this),
        f = this.scrollTop,
        d = this.scrollHeight,
        b = g.height(),
        i = h.originalEvent.wheelDelta,
        a = i > 0,
        c = function () {
          h.stopPropagation();
          h.preventDefault();
          h.returnValue = false;
          return false
        };
    if (!a && -i > d - b - f) {
      g.scrollTop(d);
      return c()
    } else {
      if (a && i > f) {
        g.scrollTop(0);
        return c()
      }
    }
  })
};

$.fn.scrollRelease = function () {
  return $(this).off("DOMMouseScroll mousewheel")
};
